Firestore rules need updating — to read all user journals, add this rule in the Firebase Console under Firestore → Rules: 
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    match /journals/{userId} {
      allow read, write: if request.auth.uid == userId;
      allow read: if request.auth.token.email == 'sparakala@gmail.com';
    }
    match /_meta/{doc} {
      allow read, write: if request.auth != null;
    }
  }
}
